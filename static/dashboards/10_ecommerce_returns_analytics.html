SQL — Return Cohort & Fraud Flag Query
Azure Synapse · Identifies serial returners and high-risk accounts
Azure Synapse Analytics
-- Identify serial returners and flag potential return fraud -- Threshold: >4 returns in 90 days OR return rate >80% of orders WITH customer_returns AS ( SELECT o.customer_id, c.email, c.registration_date, COUNT(o.order_id) AS total_orders, COUNT(r.return_id) AS total_returns, SUM(r.refund_amount) AS total_refunded, ROUND(COUNT(r.return_id) * 100.0 / NULLIF(COUNT(o.order_id),0), 1) AS return_rate_pct, MAX(r.return_date) AS last_return_date FROM orders o LEFT JOIN returns r ON o.order_id = r.order_id LEFT JOIN customers c ON o.customer_id = c.customer_id WHERE o.order_date >= DATEADD('day', -90, GETDATE()) AND o.channel IN ('web', 'app', 'wholesale') GROUP BY o.customer_id, c.email, c.registration_date ) SELECT customer_id, email, total_orders, total_returns, return_rate_pct, ROUND(total_refunded, 2) AS total_refunded, CASE WHEN return_rate_pct >= 80 AND total_returns >= 4 THEN 'HIGH RISK' WHEN return_rate_pct >= 60 THEN 'MONITOR' ELSE 'NORMAL' END AS fraud_flag, DATEDIFF('day', registration_date, GETDATE()) AS account_age_days FROM customer_returns WHERE return_rate_pct >= 60 OR total_refunded > 500 ORDER BY total_refunded DESC LIMIT 1000;
