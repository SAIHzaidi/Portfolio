The SQL — Patient Flow Analysis
EPR → SQL Server → Power BI · GDPR-compliant · pseudonymised
-- Referral-to-Treatment pathway analysis -- Pseudonymised patient_id (GDPR compliant) -- Identifies breaches and root-cause pathways WITH pathways AS ( SELECT p.pathway_id, p.patient_id, -- pseudonymised p.specialty, p.referral_date, p.treatment_date, DATEDIFF(week, p.referral_date, COALESCE(p.treatment_date, GETDATE())) AS wait_weeks, CASE WHEN DATEDIFF(week, p.referral_date, COALESCE(p.treatment_date, GETDATE())) <= 18 THEN 1 ELSE 0 END AS within_18wk, p.referral_source, p.discharge_method FROM clinical.rtt_pathways p WHERE p.referral_date >= '2024-01-01' AND p.is_complete = 1 ), delays AS ( SELECT pathway_id, delay_reason, delay_weeks, delay_stage -- diagnostic/treatment/admin FROM clinical.pathway_delays ) SELECT pw.specialty, COUNT(*) AS total_pathways, ROUND(AVG(pw.within_18wk)*100,1) AS pct_within_18wk, MAX(pw.wait_weeks) AS max_wait_weeks, AVG(pw.wait_weeks) AS avg_wait_weeks, SUM(CASE WHEN pw.within_18wk=0 THEN 1 ELSE 0 END) AS breach_count, d.delay_reason, SUM(d.delay_weeks) AS total_delay_weeks FROM pathways pw LEFT JOIN delays d USING(pathway_id) GROUP BY pw.specialty, d.delay_reason ORDER BY pct_within_18wk ASC
