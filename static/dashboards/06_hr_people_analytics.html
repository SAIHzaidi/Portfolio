The SQL — Attrition Driver Query
Workday HRIS → Redshift · weekly scheduled
-- Identify attrition predictors by segment -- Output feeds Tableau attrition risk dashboard WITH employee_base AS ( SELECT e.employee_id, e.department, e.job_level, e.tenure_months, e.salary, e.last_promotion_date, e.manager_id, DATE_DIFF(CURRENT_DATE(), e.last_promotion_date, MONTH) AS months_since_promo, CASE WHEN t.termination_date IS NOT NULL AND t.termination_type = 'Voluntary' THEN 1 ELSE 0 END AS is_attrited FROM hris.employees e LEFT JOIN hris.terminations t USING(employee_id) WHERE e.hire_date <= '2024-12-31' ), mgr_span AS ( SELECT manager_id, COUNT(*) AS direct_reports FROM employee_base GROUP BY 1 ), survey AS ( SELECT employee_id, AVG(engagement_score) AS eng_score, AVG(manager_rating) AS mgr_score FROM surveys.pulse_results WHERE survey_year = 2024 GROUP BY 1 ) SELECT e.department, e.job_level, ROUND(AVG(e.tenure_months),1) AS avg_tenure_mo, ROUND(AVG(e.months_since_promo),1) AS avg_mo_since_promo, ROUND(AVG(s.eng_score),2) AS avg_engagement, ROUND(AVG(m.direct_reports),1) AS avg_span_of_ctrl, ROUND(AVG(e.is_attrited)*100,1) AS attrition_rate_pct, COUNT(*) AS n_employees FROM employee_base e LEFT JOIN survey s USING(employee_id) LEFT JOIN mgr_span m ON e.manager_id = m.manager_id GROUP BY 1,2 HAVING COUNT(*) >= 10 ORDER BY attrition_rate_pct DESC
