The SQL — MRR Movement
Stripe → Snowflake · categorises every subscription event
-- MRR movement categorisation -- Classifies each subscription change into -- New / Expansion / Contraction / Churn / Reactivation WITH monthly_mrr AS ( SELECT customer_id, DATE_TRUNC('month', period_start) AS month, SUM(amount) / 100.0 AS mrr FROM stripe.invoice_line_items WHERE status = 'paid' GROUP BY 1,2 ), mrr_movement AS ( SELECT curr.customer_id, curr.month, curr.mrr AS current_mrr, prev.mrr AS prior_mrr, curr.mrr - COALESCE(prev.mrr,0) AS delta, CASE WHEN prev.mrr IS NULL THEN 'new' WHEN curr.mrr = 0 AND prev.mrr > 0 THEN 'churn' WHEN curr.mrr > prev.mrr THEN 'expansion' WHEN curr.mrr < prev.mrr AND curr.mrr > 0 THEN 'contraction' ELSE 'flat' END AS movement_type FROM monthly_mrr curr LEFT JOIN monthly_mrr prev ON curr.customer_id = prev.customer_id AND prev.month = DATE_TRUNC('month', DATEADD('month', -1, curr.month)) ) SELECT month, movement_type, COUNT(DISTINCT customer_id) AS accounts, ROUND(SUM(delta), 2) AS mrr_impact FROM mrr_movement WHERE movement_type != 'flat' GROUP BY 1,2 ORDER BY month, movement_type

"Too expensive" masks a value perception problem
