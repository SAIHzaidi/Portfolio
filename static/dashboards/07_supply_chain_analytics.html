The SQL — Inventory Reorder Alert Query
SAP ERP → BigQuery · runs every 4 hours · triggers Slack alert
-- Automated inventory alert system -- Identifies SKUs below reorder point -- Calculates days-of-stock from demand forecast WITH current_stock AS ( SELECT s.sku_id, s.product_name, s.category, SUM(s.quantity_on_hand) AS total_stock, SUM(s.quantity_on_order) AS on_order FROM inventory.stock_levels s WHERE s.snapshot_date = CURRENT_DATE() GROUP BY 1,2,3 ), demand_forecast AS ( SELECT sku_id, AVG(daily_demand_units) AS avg_daily_demand, STDDEV(daily_demand_units) AS demand_std FROM forecasting.demand_daily WHERE forecast_date BETWEEN DATE_SUB(CURRENT_DATE(), INTERVAL 28 DAY) AND CURRENT_DATE() GROUP BY 1 ), reorder_params AS ( SELECT sku_id, safety_stock_units, reorder_point_units, lead_time_days, supplier_id FROM procurement.sku_parameters ) SELECT c.product_name, c.category, c.total_stock, c.on_order, r.reorder_point_units, r.safety_stock_units, ROUND(c.total_stock / NULLIF(f.avg_daily_demand,0), 1) AS days_of_stock, CASE WHEN c.total_stock <= r.safety_stock_units THEN 'CRITICAL' WHEN c.total_stock <= r.reorder_point_units THEN 'REORDER NOW' WHEN c.total_stock > r.reorder_point_units * 2 THEN 'OVERSTOCKED' ELSE 'OK' END AS stock_status FROM current_stock c JOIN demand_forecast f USING(sku_id) JOIN reorder_params r USING(sku_id) ORDER BY days_of_stock ASC LIMIT 50
