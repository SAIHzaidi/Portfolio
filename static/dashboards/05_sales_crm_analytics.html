The SQL — Pipeline Velocity Query
Salesforce → BigQuery · powers this dashboard
-- Pipeline velocity: deals × win rate × ACV / cycle days -- Runs nightly via Airflow, feeds Power BI dataset WITH pipeline AS ( SELECT o.owner_name AS rep, o.lead_source, o.stage_name, o.amount AS acv, o.close_probability AS prob, DATE_DIFF(o.close_date, o.created_date, DAY) AS cycle_days, o.amount * o.close_probability / 100 AS weighted_value, CASE WHEN o.stage_name = 'Closed Won' THEN 1 ELSE 0 END AS is_won FROM salesforce.opportunities o WHERE o.fiscal_year = 2024 AND o.is_deleted = FALSE ), velocity AS ( SELECT rep, lead_source, COUNT(*) AS n_deals, ROUND(AVG(is_won)*100,1) AS win_rate_pct, ROUND(AVG(acv),0) AS avg_acv, ROUND(AVG(cycle_days),0) AS avg_cycle, SUM(weighted_value) AS weighted_pipe, -- Pipeline velocity formula ROUND( COUNT(*) * AVG(is_won) * AVG(acv) / NULLIF(AVG(cycle_days),0) ,0) AS velocity_per_day FROM pipeline GROUP BY 1,2 ) SELECT * FROM velocity ORDER BY weighted_pipe DESC
