Why Recall matters here

# churn_model.py — production classifier # Gradient Boosted Trees with calibrated threshold from sklearn.ensemble import GradientBoostingClassifier from sklearn.calibration import CalibratedClassifierCV from sklearn.pipeline import Pipeline from sklearn.preprocessing import StandardScaler from sklearn.metrics import roc_auc_score, classification_report import shap, numpy as np def build_pipeline() -> Pipeline: return Pipeline([ ('scaler', StandardScaler()), ('clf', CalibratedClassifierCV( GradientBoostingClassifier( n_estimators=400, learning_rate=0.05, max_depth=4, subsample=0.8, random_state=42, ), cv=5, method='isotonic' )) ]) def tune_threshold(model, X_val, y_val, cost_fn=840, cost_fp=12): # Find threshold minimising expected cost # rather than defaulting to 0.5 probs = model.predict_proba(X_val)[:,1] thresholds = np.arange(0.2, 0.9, 0.01) costs = [] for t in thresholds: preds = (probs >= t).astype(int) fp = ((preds==1) & (y_val==0)).sum() fn = ((preds==0) & (y_val==1)).sum() costs.append(fp * cost_fp + fn * cost_fn) best_t = thresholds[np.argmin(costs)] return best_t, np.min(costs) def explain_predictions(model, X_sample): # SHAP values for individual-level explanations # Used in the retention team's dashboard explainer = shap.TreeExplainer( model.named_steps('clf') .base_estimator ) shap_vals = explainer.shap_values(X_sample) return shap_vals
-- SQL: Build churn prediction features -- 90-day observation window per customer WITH base AS ( SELECT customer_id, DATE_TRUNC(subscription_start, MONTH) AS cohort, plan_type, mrr, contract_length_months FROM subscriptions WHERE status = 'active' ), usage AS ( SELECT customer_id, COUNT(DISTINCT session_date) AS active_days, AVG(session_duration_mins) AS avg_session, COUNT(DISTINCT feature_used) AS feature_breadth, MAX(session_date) AS last_active FROM product_events WHERE event_date >= DATE_SUB(CURRENT_DATE(), INTERVAL 90 DAY) GROUP BY 1 ), support AS ( SELECT customer_id, COUNT(*) AS ticket_count, AVG(csat_score) AS avg_csat, SUM(CASE WHEN priority='high' THEN 1 ELSE 0 END) AS high_pri_tickets FROM support_tickets WHERE created_date >= DATE_SUB(CURRENT_DATE(), INTERVAL 90 DAY) GROUP BY 1 ) SELECT b.*, u.active_days, u.avg_session, u.feature_breadth, DATE_DIFF(CURRENT_DATE(), u.last_active, DAY) AS days_since_active, s.ticket_count, s.avg_csat, s.high_pri_tickets FROM base b LEFT JOIN usage u USING(customer_id) LEFT JOIN support s USING(customer_id)

Threshold tuning saved £280K

Default 0.5 threshold optimises accuracy. Cost-weighted threshold (0.38) optimises business value — catches 74.8% of churners vs 61% at 0.5. Projected impact: £280K additional LTV retained per quarter.
